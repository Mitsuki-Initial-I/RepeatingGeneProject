using UnityEngine;

public class Environment : MonoBehaviour
{
    public float temperature = 20f;   // 気温
    public float resourceAbundance = 1f;  // 資源の豊富さ
    public bool isDay = true; // 昼夜サイクル

    private float dayNightCycleDuration = 60f;
    private float cycleTimer = 0f;

    void Update()
    {
        // 昼夜サイクルの処理
        cycleTimer += Time.deltaTime;
        if (cycleTimer >= dayNightCycleDuration)
        {
            isDay = !isDay;
            cycleTimer = 0f;
            Debug.Log(isDay ? "It's now daytime." : "It's now nighttime.");
        }

        // 環境の変化をシミュレーション
        SimulateClimate();
    }

    void SimulateClimate()
    {
        if (isDay)
        {
            temperature += 0.1f * Time.deltaTime;  // 日中は気温が上昇
        }
        else
        {
            temperature -= 0.1f * Time.deltaTime;  // 夜間は気温が低下
        }

        Debug.Log("Current temperature: " + temperature);
    }

    // 資源を提供する関数
    public float GetResourceAvailability()
    {
        return resourceAbundance;
    }
}
