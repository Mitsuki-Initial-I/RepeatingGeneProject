using UnityEngine;

public class PredatorAndPrey : MonoBehaviour
{
    public bool isPredator = false;
    public bool isPrey = false;

    void Update()
    {
        if (isPredator)
        {
            Hunt();
        }
        if (isPrey)
        {
            AvoidPredator();
        }
    }

    void Hunt()
    {
        Debug.Log("Predator is hunting.");
    }

    void AvoidPredator()
    {
        Debug.Log("Prey is avoiding predator.");
    }
}
