using System;

public enum DietType
{
    Herbivore,  // 草食動物
    Carnivore,  // 肉食動物
    Omnivore    // 雑食動物
}

public class FoodChain
{
    public static bool CanEat(Creature predator, Creature prey)
    {
        // 肉食動物は草食動物や雑食動物を食べる
        if (predator.dietType == DietType.Carnivore && (prey.dietType == DietType.Herbivore || prey.dietType == DietType.Omnivore))
        {
            return true;
        }

        // 雑食動物は草食動物を食べられるが、肉食動物には食べられる
        if (predator.dietType == DietType.Omnivore && prey.dietType == DietType.Herbivore)
        {
            return true;
        }

        // 草食動物は基本的に他の動物を食べない
        return false;
    }
}
