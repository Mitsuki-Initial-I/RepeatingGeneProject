using System;

public class Environment
{
    public float temperature = 20f;
    public float resourceAbundance = 1f;
    public bool isDay = true;

    private float dayNightCycleDuration = 60f;
    private float cycleTimer = 0f;

    public void Update(float deltaTime)
    {
        cycleTimer += deltaTime;
        if (cycleTimer >= dayNightCycleDuration)
        {
            isDay = !isDay;
            cycleTimer = 0f;
            Console.WriteLine(isDay ? "It's now daytime." : "It's now nighttime.");
        }

        SimulateEnvironment(deltaTime);
    }

    void SimulateEnvironment(float deltaTime)
    {
        if (isDay)
        {
            temperature += 0.1f * deltaTime;
        }
        else
        {
            temperature -= 0.1f * deltaTime;
        }
        Console.WriteLine("Temperature: " + temperature);
    }

    public float GetResourceAvailability()
    {
        return resourceAbundance;
    }
}
