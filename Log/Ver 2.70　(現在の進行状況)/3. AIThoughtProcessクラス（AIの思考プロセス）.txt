public class AIThoughtProcess
{
    private List<string> actions = new List<string> { "Move", "SeekFood", "Hunt", "Sleep", "DrinkWater", "Mate" };

    public void Think(Environment environment, Creature creature)
    {
        // 環境やクリーチャーの状態に基づいて行動を選択
        if (creature.hunger < 50) actions.Add("SeekFood");
        if (creature.sleepiness > 80) actions.Add("Sleep");
        // 例として簡略化。詳細なロジックを追加可能
    }

    public List<string> GetAvailableActions()
    {
        return actions;
    }
}
