using UnityEngine;

public class HealthSystem : MonoBehaviour
{
    public float health = 100f;
    public bool isSick = false;

    void Update()
    {
        if (isSick)
        {
            health -= 0.1f * Time.deltaTime;
            if (health <= 0f)
            {
                GetComponent<Creature>().Die();
            }
        }
    }

    public void GetSick()
    {
        isSick = true;
        Debug.Log("Creature is sick.");
    }

    public void Recover()
    {
        isSick = false;
        Debug.Log("Creature has recovered.");
    }
}
