using UnityEngine;

public class SocialBehavior : MonoBehaviour
{
    public float socialRank = 1f;
    public bool isLeader = false;

    public void InteractWith(Creature other)
    {
        if (isLeader)
        {
            Debug.Log("This creature is leading.");
        }
        else if (socialRank > other.socialRank)
        {
            Debug.Log("This creature is dominant.");
        }
        else
        {
            Debug.Log("This creature is submissive.");
        }
    }
}
